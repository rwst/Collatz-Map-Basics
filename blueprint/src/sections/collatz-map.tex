\begin{definition}\label{def:collatz_step}\lean{collatz_step}\leanok
The Collatz map $C: \mathbb{N} \to \mathbb{N}$ is defined by
\begin{equation}
    C(n) =
    \begin{cases}
        n/2 & \text{if } n \text{ is even}, \\
        3n + 1 & \text{if } n \text{ is odd}.
    \end{cases}
\end{equation}
\end{definition}
Lagarias reviewed the history of the map and the conjecture in \cite{lagarias}.

\begin{definition}\label{def:collatz_iter}\lean{collatz_iter}\leanok
For any $k \in \mathbb{N}$, the $k$-fold iteration of the Collatz map $C^k: \mathbb{N} \to \mathbb{N}$ is defined recursively by
\begin{equation}
    C^0(n) = n
\end{equation}
and
\begin{equation}
    C^{k+1}(n) = C^k(C(n)).
\end{equation}
\end{definition}

\begin{lemma}\label{lemma:collatz_iter_succ_right'}\lean{collatz_iter_succ_right'}\leanok
For all $k, n \in \mathbb{N}$, we have $C^{k+1}(n) = C(C^k(n))$.
\end{lemma}
\begin{proof}\leanok
We proceed by induction on $k \in \mathbb{N}$.
For the base case $k=0$, $C^1(n) = C(C^0(n)) = C(n)$ by definition.
For the inductive step $k+1$, we have:
\begin{align*}
    C^{k+2}(n) &= C^{k+1}(C(n)) \tag{by definition} \\
    &= C(C^k(C(n))) \tag{by induction hypothesis applied to $C(n)$} \\
    &= C(C^{k+1}(n)) \tag{by definition.}
\end{align*}
\end{proof}

\begin{lemma}\label{lemma:collatz_step_zero}\lean{collatz_step_zero}\leanok
$C(0) = 0$.
\end{lemma}
\begin{proof}\leanok
Trivial by definition.
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_zero}\lean{collatz_iter_zero}\leanok
For all $k \in \mathbb{N}$, $C^k(0) = 0$.
\end{lemma}
\begin{proof}\leanok
By induction on $k$.
\end{proof}

\begin{lemma}\label{lemma:collatz_step_pos}\lean{collatz_step_pos}\leanok
If $n \ge 1$, then $C(n) \ge 1$.
\end{lemma}
\begin{proof}\leanok
If $n$ is even, then $n \ge 2$, so $n/2 \ge 1$. If $n$ is odd, then $3n+1 \ge 4 \ge 1$.
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_pos}\lean{collatz_iter_pos}\leanok
If $n \ge 1$, then for all $k \in \mathbb{N}$, $C^k(n) \ge 1$.
\end{lemma}
\begin{proof}\leanok
By induction on $k$, using Lemma~\ref{lemma:collatz_step_pos}.
\end{proof}

\begin{lemma}\label{lemma:collatz_step_pow_two}\lean{collatz_step_pow_two}\leanok
For all $k \ge 1$, $C(2^k) = 2^{k-1}$.
\end{lemma}
\begin{proof}\leanok
Since $2^k$ is even for $k \ge 1$, $C(2^k) = 2^k/2 = 2^{k-1}$.
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_two_pow_mul}\lean{collatz_iter_two_pow_mul}\leanok
For any $k, m \in \mathbb{N}$, $C^k(2^k \cdot m) = m$.
\end{lemma}
\begin{proof}\leanok
We proceed by induction on $k \in \mathbb{N}$. The base case $k=0$ is trivial as $C^0(m) = m$.
For the inductive step $k+1$, we have
\begin{align*}
    C^{k+1}(2^{k+1} \cdot m) &= C^k(C(2 \cdot 2^k \cdot m)) \\
    &= C^k(2^k \cdot m) \tag{since $2 \cdot 2^k \cdot m$ is even} \\
    &= m \tag{by induction hypothesis.}
\end{align*}
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_pow_two}\lean{collatz_iter_pow_two}\leanok
For all $l \in \mathbb{N}$, $C^l(2^l) = 1$.
\end{lemma}
\begin{proof}\leanok
By induction on $l$, using Lemma~\ref{lemma:collatz_step_pow_two}.
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_pow_two_ne_one}\lean{collatz_iter_pow_two_ne_one}\leanok
For all $k < l$, $C^k(2^l) \neq 1$.
\end{lemma}
\begin{proof}\leanok
By induction on $k$, using Lemma~\ref{lemma:collatz_step_pow_two}.
\end{proof}

\begin{lemma}\label{lemma:exists_collatz_reverse_strict}\lean{exists_collatz_reverse_strict}\leanok
For all $l \in \mathbb{N}$, there exists $n \in \mathbb{N}$ such that $C^l(n) = 1$ and for all $k < l$, $C^k(n) \neq 1$.
\end{lemma}
\begin{proof}\leanok
Choose $n = 2^l$ and apply Lemmas~\ref{lemma:collatz_iter_pow_two} and \ref{lemma:collatz_iter_pow_two_ne_one}.
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_add}\lean{collatz_iter_add}\leanok
For all $a, b, n \in \mathbb{N}$, $C^{a+b}(n) = C^a(C^b(n))$.
\end{lemma}
\begin{proof}\leanok
By induction on $b$.
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_mul_cycle}\lean{collatz_iter_mul_cycle}\leanok
If $C^k(n) = n$, then for all $m \in \mathbb{N}$, $C^{m \cdot k}(n) = n$.
\end{lemma}
\begin{proof}\leanok
By induction on $m$, using Lemma~\ref{lemma:collatz_iter_add}.
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_mem_124}\lean{collatz_iter_mem_124}\leanok
If $n \in \{1, 2, 4\}$, then for all $i \in \mathbb{N}$, $C^i(n) \in \{1, 2, 4\}$.
\end{lemma}
\begin{proof}\leanok
By induction on $i$, checking the cases $n=1, 2, 4$ for $C(n)$.
\end{proof}

\begin{lemma}\label{lemma:collatz_iter_one_le_four}\lean{collatz_iter_one_le_four}\leanok
For all $i \in \mathbb{N}$, $C^i(1) \le 4$.
\end{lemma}
\begin{proof}\leanok
Since $1 \in \{1, 2, 4\}$, we have $C^i(1) \in \{1, 2, 4\}$ by Lemma~\ref{lemma:collatz_iter_mem_124}, so $C^i(1) \le 4$.
\end{proof}

\begin{lemma}\label{lemma:collatz_step_closed_form}\lean{collatz_step_closed_form}\leanok
The Collatz map $C(n)$ has the closed form
\begin{equation}
    C(n) = \frac{(7n + 2) - (5n + 2)(-1)^n}{4}
\end{equation}
for all $n \in \mathbb{N}$.
\end{lemma}
\begin{proof}\leanok
We split into cases based on the parity of $n$.
If $n$ is even, then $(-1)^n = 1$, so
\begin{equation}
    \frac{(7n + 2) - (5n + 2)(1)}{4} = \frac{2n}{4} = \frac{n}{2} = C(n).
\end{equation}
If $n$ is odd, then $(-1)^n = -1$, so
\begin{equation}
    \frac{(7n + 2) - (5n + 2)(-1)}{4} = \frac{12n + 4}{4} = 3n + 1 = C(n).
\end{equation}
In both cases, the formula yields $C(n)$.
\end{proof}
